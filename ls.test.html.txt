    const questions = [

        { q: "你做决定时的习惯是：", a: "果断干脆，相信自己的第一直觉。", b: "反复权衡利弊，收集尽可能多的信息后再定夺。", c: "很难下定决心，常常想听听身边人的建议。" },
        { q: "当你和别人发生意见分歧时，你会：", a: "坚持自己的观点，直到说服对方。", b: "寻求折中方案，大家都退让一步。", c: "避免正面争吵，甚至可能为了和谐而妥协。" },

        { q: "关于人生目标，你的态度倾向于：", a: "制定详细的长期计划并努力实现。", b: "设立短期可行的目标，走好每一步。", c: "顺其自然，看重享受当下。" }

    const resultsData = {
        A: { title: "“行动派领导者”", text: "你是一个果断干脆、充满活力的人。你喜欢掌控局势，能够迅速做出决定，并在压力下表现出很强的执行力。你更看重未来的可能性，是个天生的领导者。" },
        B: { title: "“谨慎派专家”", text: "你是一个心思细腻、注重逻辑和细节的人。你做决定前喜欢收集足够的信息，反复权衡利弊。你享受独处和思考，是个踏实的执行者。" },
        C: { title: "“随和派协调者”", text: "你是一个平和、友善且懂得享受当下的人。你善于协调人际关系，避免冲突，是团队中的“润滑剂”。你让人感到舒服，懂得平衡生活。" }
    };

    let currentIndex = 0;
    let userAnswers = [];

    function showQuestion() {
        const q = questions[currentIndex];
        document.getElementById('question-text').innerText = q.q;
        document.getElementById('step-info').innerText = `第 ${currentIndex + 1} / ${questions.length} 题`;
        document.getElementById('progress-bar').style.width = ((currentIndex + 1) / questions.length * 100) + '%';
        
        const optionsList = document.getElementById('options-list');
        optionsList.innerHTML = '';
        
        // 生成三个选项按钮
        ['A', 'B', 'C'].forEach(type => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.innerText = q[type.toLowerCase()];
            btn.onclick = () => selectOption(type);
            optionsList.appendChild(btn);
        });

        // 控制“上一题”按钮显示
        document.getElementById('prev-btn').style.visibility = currentIndex === 0 ? 'hidden' : 'visible';
    }

    function selectOption(type) {
        userAnswers[currentIndex] = type;
        if (currentIndex < questions.length - 1) {
            currentIndex++;
            setTimeout(showQuestion, 200); // 稍微延迟一下，让用户看清点击效果
        } else {
            showFinalResult();
        }
    }

    function goPrev() {
        if (currentIndex > 0) {
            currentIndex--;
            showQuestion();
        }
    }

    function showFinalResult() {
        const counts = { A: 0, B: 0, C: 0 };
        userAnswers.forEach(ans => counts[ans]++);
        
        let maxType = 'A';
        if (counts.B > counts[maxType]) maxType = 'B';
        if (counts.C > counts[maxType]) maxType = 'C';

        document.getElementById('quiz-card').classList.add('hidden');
        const resCard = document.getElementById('result-card');
        resCard.classList.remove('hidden');
        
        document.getElementById('res-title').innerText = `结果：${resultsData[maxType].title}`;
        document.getElementById('res-text').innerText = resultsData[maxType].text;
    }

    // 初始化
    window.onload = showQuestion;
</script>
